function validasiPendaftaran() {
    const nama = document.getElementById('nama').value.trim();
    const unit = document.getElementById('unit').value.trim();
    const kategori = document.getElementById('kategori').value;
    const jenis = document.getElementById('jenis').value;
    const tglInput = document.getElementById('tglLahir').value;
    const resultDiv = document.getElementById('result');
    const btnDaftar = document.getElementById('linkDaftar');

    // Pemetaan Link sesuai permintaan
    const links = {
        "FASI": {
            "PERORANGAN": "https://forms.gle/kpwFuToc5gdX3sU38",
            "GROUP": "https://forms.gle/hDcr8RTVtX7AVqzv5"
        },
        "GUNTARA": {
            "PERORANGAN": "https://forms.gle/Ecf8Qmg5tWgFMfCC8",
            "GROUP": "https://forms.gle/v8tDH9KnTvoUWfpW6"
        }
    };

    if (!nama || !unit || !tglInput) {
        alert("Harap lengkapi data Nama, Unit, dan Tanggal Lahir!");
        return;
    }

    const tglLahir = new Date(tglInput);
    const tglCutOff = new Date('2026-07-01');

    let thn = tglCutOff.getFullYear() - tglLahir.getFullYear();
    let bln = tglCutOff.getMonth() - tglLahir.getMonth();
    let hr = tglCutOff.getDate() - tglLahir.getDate();

    if (hr < 0) {
        bln--;
        const lastDayPrevMonth = new Date(tglCutOff.getFullYear(), tglCutOff.getMonth(), 0).getDate();
        hr += lastDayPrevMonth;
    }
    if (bln < 0) {
        thn--;
        bln += 12;
    }

    let min = 0, max = 0, labelKategori = "", grupInduk = "";
    if (kategori === "TKA") { min = 0; max = 7; labelKategori = "FASI TKA"; grupInduk = "FASI"; }
    else if (kategori === "TPA") { min = 7; max = 12; labelKategori = "FASI TPA"; grupInduk = "FASI"; }
    else if (kategori === "TQA") { min = 12; max = 15; labelKategori = "FASI TQA"; grupInduk = "FASI"; }
    else if (kategori === "GUNTARA") { min = 18; max = 45; labelKategori = "GUNTARA"; grupInduk = "GUNTARA"; }

    resultDiv.style.display = "block";

    const sudahLewatBatasMaks = (thn > max || (thn === max && (bln > 0 || hr > 0)));
    const belumCukupMin = (thn < min || (thn === min && bln === 0 && hr === 0 && min !== 0)); 

    if (belumCukupMin) {
        resultDiv.className = "error";
        resultDiv.innerHTML = `⚠️ <strong>DITOLAK: TERLALU MUDA</strong><br><small>Umur Anda ${thn} Thn. Minimal untuk ${labelKategori} adalah ${min} Tahun.</small>`;
        btnDaftar.style.display = "none";
    } else if (sudahLewatBatasMaks) {
        resultDiv.className = "error";
        resultDiv.innerHTML = `⚠️ <strong>DITOLAK: MELEWATI BATAS</strong><br><small>Umur Anda ${thn} Thn ${bln} Bln. Maksimal untuk ${labelKategori} adalah ${max} Tahun.</small>`;
        btnDaftar.style.display = "none";
    } else {
        resultDiv.className = "success";
        resultDiv.innerHTML = `✅ <strong>LOLOS VERIFIKASI UMUR</strong><br><small>Halo ${nama}, data Anda sesuai untuk kategori <b>${labelKategori} (${jenis})</b>.</small>`;
        
        // Pilih Link berdasarkan Grup Induk (FASI/GUNTARA) dan Jenis (Perorangan/Group)
        btnDaftar.href = links[grupInduk][jenis];
        btnDaftar.style.display = "block";
    }
}
</script>
